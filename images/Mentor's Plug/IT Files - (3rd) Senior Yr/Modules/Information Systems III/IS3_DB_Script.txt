 DROP TABLE CUSTOMERS;
DROP TABLE ITEMS;
DROP TABLE COURIER;
DROP TABLE ORDERS;


CREATE TABLE CUSTOMERS
 ( CUSTOMERID INT(6) PRIMARY KEY,
   FIRSTNAME VARCHAR(20),
   SURNAME VARCHAR(20),
   CUSTPHONE VARCHAR(16),
   CUSTADD VARCHAR(30));

CREATE TABLE ITEMS
( ITEMID INT(4) PRIMARY KEY,
  COLOUR VARCHAR(20),
  WEIGHT DOUBLE(3,2),
  DIMENSIONS VARCHAR(40),
  PRICE DOUBLE(6,2));

CREATE TABLE COURIER
( COURIERID INT(6) PRIMARY KEY,
  COURIERNAME VARCHAR(30),
  COURIERPHONE VARCHAR(16),
  EMAIL VARCHAR(30));

CREATE TABLE ORDERS
( ORDERID INT(6) PRIMARY KEY,
  ORDERDATE DATE,
  ITEMID INT(4),
  COURIERID INT(6),
  CUSTOMERID INT(6),
  QTY_ORDERED INT(4),
  CONSTRAINT FK_ITEM FOREIGN KEY (ITEMID) REFERENCES ITEMS(ITEMID),
  CONSTRAINT FK_COURIER FOREIGN KEY(COURIERID) REFERENCES COURIER(COURIERID),
  CONSTRAINT FK_CUSTOMER FOREIGN KEY(CUSTOMERID) REFERENCES CUSTOMERS(CUSTOMERID));
